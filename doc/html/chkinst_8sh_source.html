58 ## @fn differ()
59 ## @brief Display diff of specified files and update if needed
60 ## @param param1 first argument is file in current directory
61 ## @param param2 second argument is installed version of file
62 differ() {
63  echo "$1 differs from installed version $2"
64  echo ""
65  [ "$DIFF" ] && {
66  diff "$1" "$2"
67  echo ""
68  }
69  [ "$UPD" ] && {
70  REFORCE=
71  FOUND=
72  for cmd in $DIFFER
73  do
74  [ "$1" = "$cmd" ] && {
75  FOUND=1
76  break
77  }
78  done
79  [ "$FOUND" ] && {
80  echo "Looks like you need to update one of your customized files."
81  echo "$1 is on your list of files to prompt before installing."
82  [ "$FORCE" ] && {
83  FORCE=
84  REFORCE=1
85  }
86  }
87  if [ "$FORCE" ]
88  then
89  update "$1" "$2"
90  else
91  while true
92  do
93  read -p "Do you want to install the new version ? (y/n) " yn
94  case $yn in
95  [Yy]* ) update "$1" "$2"; break;;
96  [Nn]* ) break;;
97  * ) echo "Please answer yes or no.";;
98  esac
99  done
100  [ "$REFORCE" ] && FORCE=1
101  fi
102  }
103 }
